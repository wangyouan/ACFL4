\begin{frame}{Reducing Costs or Increasing Subsidies: A Financing Channel}
	Certain financing structures offer explicit or implicit subsidies.
	\begin{block}{Illustration: Subsidized Bond Issuance}
		Ant Sports Co. applies for a tax-exempt, 5-year, \textyen 2 million industrial bond in Fujian to avoid relocating production to Vietnam.
		\begin{itemize}
			\item Regular debt cost: 10\%
			\item Industrial bond coupon: 5\%
			\item Annual interest savings: \textyen 100,000
		\end{itemize}
		\textbf{Question}: What is the NPV of accepting the subsidy?
	\end{block}
\end{frame}

\begin{frame}{Valuation of Subsidized Financing}
	\textbf{NPV Calculation}:
	\begin{align*}
		NPV &= \text{\textyen} 2,000,000 - \sum_{t=1}^5 \frac{\text{\textyen} 100,000}{1.1^t} - \frac{\text{\textyen} 2,000,000}{1.1^5} \\
		&= \text{\textyen} 2,000,000 - \text{\textyen} 1,620,921 \\
		&= \text{\textyen} 379,079
	\end{align*}
	\textbf{Conclusion}:  
	Ant Sports captures a \textbf{subsidy} worth \textyen 379,079 by issuing the industrial bond.

 \textbf{Interpretation}: The Fujian government provides an implicit subsidy equal to NPV via below-market rates
\end{frame}

\subsection{Efficient Capital Markets}

\begin{frame}{Historical Foundations of the Efficient Market Hypothesis}
	
	\begin{columns}[T]
		\begin{column}{0.52\textwidth}
			\textbf{Foundational Milestones:}
			\begin{itemize}
				\item \textbf{1900}: \textcite{bachelier1900} introduces probabilistic asset price modeling
				\item \textbf{1965}: \textcite{fama1965random} presents empirical support for price randomness
				\item \textbf{1970}: \textcite{fama1970efficient} formalizes EMH and classifies it into three forms
			\end{itemize}
		\end{column}
		
		\begin{column}{0.44\textwidth}
			\textbf{Further Contributions:}
			\begin{itemize}
				\item \textbf{1970s–80s}:
				\begin{itemize}
					\item Paul Samuelson — theoretical validation
					\item \textcite{jensen1978anomalous} on performance evaluation under EMH
					\item \textcite{malkiel1973random} popularizes EMH among retail investors
				\end{itemize}
				\item \textbf{2013}: Fama receives Nobel Prize for EMH development
			\end{itemize}
		\end{column}
	\end{columns}
	
	\vspace{4mm}
	\begin{block}{Key Milestones Timeline}
		1900 (Bachelier) $\rightarrow$ 1965 (Fama) $\rightarrow$ 1970 (Formalization) $\rightarrow$ 2013 (Nobel Recognition)
	\end{block}
	
\end{frame}

\begin{frame}{Understanding Efficient Capital Markets}
	
	\begin{block}{\textbf{Definition (Efficient Market Hypothesis)}}
		A financial market is said to be \textbf{efficient} if security prices \alert{fully and instantaneously reflect all available information}.
	\end{block}
	
	\vspace{0.4cm}
	
	\begin{itemize}
		\item \textbf{Mechanism:} Prices adjust \textit{immediately} to new information — no predictable patterns remain.
		\vspace{0.3cm}
		\item \textbf{Implications:}
		\begin{itemize}
			\item[$\triangleright$] \textbf{Investors:} Cannot systematically outperform the market using public information
			\item[$\triangleright$] \textbf{Firms:} Securities are fairly priced — cost of capital reflects fundamentals
			\item[$\triangleright$] \textbf{Regulators:} Focus shifts to transparency and timely disclosure
		\end{itemize}
	\end{itemize}
	
\end{frame}

\begin{frame}{Foundations Underlying Market Efficiency}
	
	\begin{enumerate}
		\item \textbf{Rational Investors}
		\begin{itemize}
			\item Agents act to maximize expected utility
			\item Beliefs updated via Bayes' rule upon arrival of new information
		\end{itemize}
		
		\vspace{0.2cm}
		\item \textbf{Limited Impact of Irrationality}
		\begin{itemize}
			\item Behavioral biases exist but are uncorrelated and cancel in aggregate
			\item Bounded rationality does not lead to systematic mispricing
		\end{itemize}
		
		\vspace{0.2cm}
		\item \textbf{Presence of Arbitrage Forces}
		\begin{itemize}
			\item Rational arbitrageurs exploit mispricing opportunities
			\item Their trading ensures prices return to fundamental values
		\end{itemize}
	\end{enumerate}
	
	\vspace{0.3cm}
	\begin{alertblock}{Key Insight}
		Market efficiency does not require every investor to be rational — only that \textit{irrationality is limited and arbitraged away}.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Forms of Market Efficiency: Nested Information Sets}
	
	\begin{columns}
		\column{0.4\textwidth}
		\includegraphics[width=\linewidth]{images/section1/informationset.png}
		
		\column{0.6\textwidth}
		\textbf{Three Forms of EMH:}
		\begin{itemize}
			\item \textbf{Weak Form}: Prices reflect all \textit{past price and volume information}
			\begin{itemize}
				\item $\rightarrow$ Technical analysis is ineffective
			\end{itemize}
			
			\item \textbf{Semi-Strong Form}: Prices reflect all \textit{publicly available information}
			\begin{itemize}
				\item $\rightarrow$ Fundamental analysis cannot yield abnormal returns
			\end{itemize}
			
			\item \textbf{Strong Form}: Prices reflect \textit{all information}, including private or insider info
			\begin{itemize}
				\item $\rightarrow$ Even insiders cannot consistently beat the market
			\end{itemize}
		\end{itemize}
	\end{columns}
	
	\begin{block}{Interpretation}
		\small
		Each broader form of efficiency \textit{subsumes} the weaker one — the informational assumptions become stronger as we move outward in the diagram.
	\end{block}
	
\end{frame}

	\begin{table}[h]
		\centering

	\vspace{0.3cm}
	\begin{block}{Key Takeaway}
		The forms of EMH are nested: each stronger form implies the weaker ones. Strong-form efficiency sets a high bar — rarely met in reality.
	\end{block}
	
\end{frame}


\begin{frame}{Common Misconceptions about Market Efficiency}
	
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\textbf{Misconceptions}
			\begin{itemize}
				\item \textbf{“Markets are random”} — implies chaos or lack of structure
				\item \textbf{“No one can make money”} — implies inefficiency punishes skill
				\item \textbf{“Prices have no cause”} — implies disconnection from fundamentals
			\end{itemize}
		\end{column}
		
		\begin{column}{0.5\textwidth}
			\textbf{Clarifications}
			\begin{itemize}
				\item \textbf{EMH $\neq$ randomness}: Prices follow a stochastic process but reflect information
				\item \textbf{EMH $\neq$ zero profit}: Investors earn risk-adjusted expected returns
				\item \textbf{EMH $\neq$ no causality}: Prices respond to new information efficiently
			\end{itemize}
		\end{column}
	\end{columns}
	
	\vspace{4mm}
	\begin{block}{Key Insight}
		\small
		The Efficient Market Hypothesis does not deny structure or rationality — it asserts that \textbf{information is rapidly and correctly incorporated into prices}.
	\end{block}
	
\end{frame}

\begin{frame}{Are Stock Price Changes Predictable?}
	
	\begin{block}{Cognitive Illusion in Markets}
		Humans are pattern-seeking creatures. We often perceive structure even in random sequences.
	\end{block}
	
	\begin{itemize}
		\item Analysts and traders frequently claim to “see” trends in price charts.
		\item But many such patterns may be statistical noise — akin to optical illusions.
		\item Even if short-term patterns exist, they must yield risk-adjusted returns net of transaction costs.
	\end{itemize}
	
	\vspace{0.3cm}
	\begin{alertblock}{Implication}
		Apparent predictability does not equal exploitable inefficiency.

\subsubsection{Event Studies}
\begin{frame}{Event Studies: What and Why}
	
	\begin{block}{Definition}
		Event studies evaluate how quickly and accurately stock prices incorporate new public information — a test of \textbf{semi-strong form efficiency}.
	\end{block}
	
	\vspace{0.2cm}
	\begin{itemize}
		\item Examine returns around the timing of a clearly defined event (e.g., earnings release).
		\item Look for price patterns indicating underreaction, overreaction, or delayed adjustment.
		\item If markets are efficient, abnormal returns should be concentrated around the event day and quickly dissipate.
	\end{itemize}
	
	\vspace{0.2cm}
	\begin{alertblock}{Key Assumption}
		Markets react rationally and rapidly to publicly available news.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Calculating Abnormal Returns}
	
	\begin{itemize}
		\item \textbf{Goal:} Isolate the effect of the event from market-wide movements.
		\item \textbf{Method 1: Market-Adjusted Model}
		\[
		AR_t = R_t - R_{M,t}
		\]
		
		\item \textbf{Method 2: Market Model (preferred)}
		\[
		AR_t = R_t - (\alpha + \beta R_{M,t})
		\]
		\item $\alpha$ and $\beta$ are estimated over a pre-event estimation window.
	\end{itemize}
	
	\vspace{0.2cm}
	\begin{block}{Interpretation}
		If $AR_t$ is statistically significant around the event, it implies market misreaction or delayed adjustment.
	\end{block}
	
\end{frame}

\begin{frame}{Event Study Evidence}
	
	\begin{itemize}
		\item Widely applied to study market reactions to:
		\begin{itemize}
			\item Earnings announcements \parencite{ball1968}
			\item Dividend changes \parencite{fama1969adjustment,michaely1995}
			\item Mergers \& acquisitions \parencite{jensen1983,andrade2001}
			\item Stock splits or repurchases \parencite{ikenberry1995}
			\item Seasoned equity offerings \parencite{loughran1995,asquith1986}
		\end{itemize}
		
		\item \textbf{General Findings}:
		\begin{itemize}
			\item Abnormal returns tend to cluster tightly around the event
			\item Markets typically adjust within minutes to hours
			\item Some evidence of anticipation (i.e., price moves before public release)
		\end{itemize}
	\end{itemize}
	
	\vspace{0.2cm}
	\begin{alertblock}{Conclusion}
		Event studies strongly support the semi-strong form of the EMH: public information is quickly reflected in prices.
	\end{alertblock}

\subsubsection{Mutual Fund Performance}
\begin{frame}{Do Active Mutual Funds Beat the Market?}
	
	\begin{block}{EMH Prediction}
		Under the \textbf{semi-strong form} of market efficiency, no investor using publicly available information should consistently earn excess risk-adjusted returns.
	\end{block}
	\small
	\begin{itemize}
		\item Actively managed mutual funds employ analyst skill and public data to select “undervalued” stocks.
		\item Benchmark: Passive market index (e.g., S\&P 500 or Vanguard 500 Index Fund).
		\item Empirical tests compare long-term returns (net of fees) between active funds and index performance.
		\item Evidence shows: \textbf{most active funds underperform} their benchmark after costs.
	\end{itemize}
	
	\begin{alertblock}{Implication}
		\footnotesize
		Consistent underperformance of active funds provides strong support for semi-strong form efficiency.
	\end{alertblock}
	
\end{frame}

\subsubsection{EMH in China}
\begin{frame}{EMH in China: Weak-Form Efficiency}
	
	\begin{block}{What is being tested?}
		Weak-form efficiency implies that stock returns should not be predictable based on past price or volume information.
	\end{block}
	
	\vspace{0.3cm}
	\textbf{Evidence from Chinese Markets:}
	\begin{itemize}
		\item Early studies using autocorrelation and runs tests:
		\begin{itemize}
			\item \textcite{laurence1997weak}, \textcite{liu1997market} find weak evidence of autocorrelation in Shanghai and Shenzhen markets.
			\item Suggest limited weak-form efficiency in early years.
		\end{itemize}
		
		\item Contrasting results:
		\begin{itemize}
			\item \textcite{mookerjee1999efficiency}, \textcite{ma2004testing} report significant serial dependence, rejecting weak-form EMH.
		\end{itemize}
	\end{itemize}
	
	\vspace{0.2cm}
	\begin{alertblock}{Conclusion}
		The weak-form efficiency of Chinese markets remains inconclusive and time-varying.
	\end{alertblock}
	
\end{frame}


\begin{frame}{Challenges in Testing EMH in China}
	
	\begin{itemize}
		\item Conflicting findings may stem from:
		\begin{itemize}
			\item Short sample periods and market immaturity
			\item Infrequent trading (“thin trading”) biases autocorrelation tests
		\end{itemize}
		
		\item \textcite{groenewold2004market} suggest apparent predictability may reflect structural market issues, not inefficiency.
		
		\item \textcite{zhang2008variance} and \textcite{charles2009variance} apply variance ratio (VR) tests with mixed outcomes.
	\end{itemize}
	
	\vspace{0.3cm}
	\begin{block}{Interpretation}
		In emerging markets like China, market microstructure effects (e.g., liquidity, institutional reform) complicate efficiency testing.
	\end{block}
	
\end{frame}

\begin{frame}{Toward Adaptive Efficiency in China}
	
	\begin{itemize}
		\item Some researchers propose that efficiency in China is \textit{time-varying} or \textit{adaptive}.
		
		\item Supporting evidence:
		\begin{itemize}
			\item As reforms progressed, return predictability weakened (suggesting increasing efficiency)
			\item VR and time-varying parameter models detect gradual evolution toward weak-form efficiency
		\end{itemize}
		
		\item \textcite{lim2011evolution} emphasize that efficiency should be viewed as evolving with market development.
	\end{itemize}
	
	\vspace{0.3cm}
	\begin{alertblock}{Takeaway}
		China’s markets may be better understood under an \textbf{adaptive efficiency} framework rather than a static EMH view.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Mutual Fund Performance in China: Overview}
	\begin{itemize}
		\item Since 1998, China's mutual fund industry has grown rapidly, reaching over RMB 1 trillion AUM by 2012.
		\item \textbf{\textcite{chi2013china}} documents:
		\begin{itemize}
			\item Strong evidence of positive alpha among Chinese mutual funds (up to 9.6\% annually gross)
			\item Significant skill in stock-picking, confirmed by bootstrap simulations
			\item Outperformance mostly driven by holdings rather than timing
		\end{itemize}
	\end{itemize}
	
	\vspace{4mm}
	\begin{alertblock}{EMH Implication}
		Results suggest improving semi-strong form efficiency—mutual fund outperformance has diminished over time.
	\end{alertblock}
\end{frame}

\subsection{Challenges for EMH}
\begin{frame}{Challenges to the EMH: Foundations Under Scrutiny}
	
	\begin{block}{Three Core Assumptions of EMH}
		The Efficient Market Hypothesis relies on the following key foundations:
		\begin{enumerate}
			\item \textbf{Rational Investors} \\
			Investors process information logically and make utility-maximizing decisions.
			
			\item \textbf{Independent Deviations from Rationality} \\
			If irrational behavior exists, it is assumed to be random and uncorrelated across individuals.
			
			\item \textbf{Unlimited and Effective Arbitrage} \\
			Rational traders can quickly correct mispricings through arbitrage activity.
		\end{enumerate}
	\end{block}
	
	\begin{alertblock}{Behavioral and Empirical Challenges}
		\small
		Behavioral finance questions each of these assumptions — demonstrating systematic biases, correlated misbeliefs, and limited arbitrage in real markets.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Foundations of EMH – Under Challenge}
	\centering
	\begin{tikzpicture}[node distance=0.5cm]
		\node (pillar1) [rounded corners, fill=blue!15, text width=3.5cm, align=center] {

		};
		\node (pillar2) [right=of pillar1, rounded corners, fill=green!15, text width=3.5cm, align=center] {
			\textbf{Independent Irrationality} \\

			\tiny \textcite{barberis1998model}, \textcite{banerjee1992simple}
		};
		\node (pillar3) [right=of pillar2, rounded corners, fill=red!15, text width=3.5cm, align=center] {
			\textbf{Unlimited Arbitrage} \\

			\tiny \textcite{shleifer1997limits}

	\vspace{0.5cm}
	\begin{itemize}
		\item \textcolor{blue}{Psychological}: Human behavior systematically deviates from rationality
		\item \textcolor{green}{Correlated Errors}: Biases are not random, but often aggregate and amplify
		\item \textcolor{red}{Implementation Barriers}: Arbitrage is risky, costly, and capital-constrained
	\end{itemize}
\end{frame}

\subsubsection{Rationality}
\begin{frame}{Challenges to Rationality in Investor Behavior}
	
	\begin{block}{Behavioral Biases Undermining Rationality}
		Traditional EMH assumes that investors process information logically and make optimal decisions. Behavioral finance identifies systematic deviations from this assumption:
	\end{block}
	
	\vspace{0.3cm}
	\begin{columns}
		\begin{column}{0.48\textwidth}
			\textbf{Cognitive Biases}
			\begin{itemize}
				\item Overconfidence
				\item Representativeness (Overreaction)
				\item Conservatism (Underreaction)
			\end{itemize}
		\end{column}
		
		\begin{column}{0.48\textwidth}
			\textbf{Emotional and Contextual Biases}
			\begin{itemize}
				\item Disposition Effect
				\item Familiarity and Home Bias
				\item Risk-Taking After Gains or Losses
			\end{itemize}
		\end{column}
	\end{columns}
	
	\vspace{0.3cm}
	\begin{alertblock}{Key Insight}
		Investor psychology introduces predictable, systematic departures from rationality — inconsistent with EMH.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Overconfidence Bias}
	
	\begin{block}{Definition}  
		Overconfidence refers to the tendency to overestimate one's own abilities, knowledge, or predictive accuracy.
	\end{block}
	
	\vspace{0.3cm}
	\textbf{Behavioral Manifestation}
	\begin{itemize}
		\item Investors believe their analysis is superior to others
		\item Excessive trading and portfolio turnover
		\item Neglect of contradictory evidence
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Market Consequences}
	\begin{itemize}
		\item Increased volatility
		\item Lower net returns after fees
		\item Mispricing due to overreaction
	\end{itemize}
	
	\vspace{0.2cm}
	\tiny{\textcite{odean1998volume}, \textcite{barber2001boys}}
\end{frame}

\begin{frame}{Disposition Effect}
	
	\begin{block}{Definition}  
		Investors tend to sell winning stocks too early and hold losing stocks too long.
	\end{block}
	
	
	\vspace{0.3cm}
	\textbf{Behavioral Mechanism}
	\begin{itemize}
		\item Selling winners brings pride ("I was right")
		\item Holding losers avoids admitting mistakes (regret aversion)
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Market Consequences}
	\begin{itemize}
		\item Return drag from suboptimal realization patterns
		\item Reversal patterns contradicting momentum
		\item Tax inefficiencies: paying gains, deferring losses
	\end{itemize}
	
	\vspace{0.2cm}
	\tiny{\textcite{shefrin1985disposition}}
\end{frame}

\begin{frame}{Familiarity and Home Bias}
	
	\begin{block}{Definition}  
		Investors tend to overweight assets they are familiar with — such as domestic markets or well-known companies.
	\end{block}
	
	
	\vspace{0.3cm}
	\textbf{Behavioral Mechanism}
	\begin{itemize}
		\item Preference for known over unknown ("ambiguity aversion")
		\item Loyalty to local or national institutions
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Market Consequences}
	\begin{itemize}
		\item Under-diversification of portfolios
		\item Missed opportunities in global markets
		\item Potential mispricing of familiar stocks
	\end{itemize}
	
	\vspace{0.2cm}
	\tiny{\textcite{huberman2001familiarity}}
\end{frame}

\begin{frame}{Representativeness Heuristic (Overreaction)}
	\begin{block}{Definition}  
		Investors incorrectly assume that recent trends are representative of long-run outcomes.
	\end{block}
	
	
	\vspace{0.3cm}
	\textbf{Behavioral Mechanism}
	\begin{itemize}
		\item Over-extrapolation from short-term patterns
		\item Ignoring base rates and statistical reasoning
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Market Consequences}
	\begin{itemize}
		\item Overreaction to earnings surprises and market news
		\item Formation of price bubbles and crashes
		\item Reversal in long-term stock returns
	\end{itemize}
	
	\vspace{0.2cm}
	\tiny{\textcite{kahneman1972subjective}, \textcite{barberis1998model}}
\end{frame}

\begin{frame}{Conservatism Bias (Underreaction)}
	\begin{block}{Definition}  
		Investors update their beliefs too slowly in response to new information.
	\end{block}
	
	\vspace{0.3cm}
	\textbf{Behavioral Mechanism}
	\begin{itemize}
		\item Anchoring on prior beliefs
		\item Cognitive inertia and inattentiveness
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Market Consequences}
	\begin{itemize}
		\item Drift in stock prices after earnings announcements
		\item Predictable return patterns over medium horizon
		\item Opportunities for momentum strategies
	\end{itemize}
	
	\vspace{0.2cm}
	\tiny{\textcite{barberis1998model}, \textcite{hong1999underreaction}}
\end{frame}

\begin{frame}{Risk-Taking After Gains or Losses}
	\begin{block}{Definition}  
		Investor risk preferences are influenced by recent portfolio performance.
	\end{block}
	
	\vspace{0.3cm}
	\textbf{Behavioral Mechanism}
	\begin{itemize}
		\item \textbf{House Money Effect}: Investors take more risk after gains
		\item \textbf{Snakebite Effect}: Investors become overly risk-averse after losses
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Market Consequences}
	\begin{itemize}
		\item Cyclical over- and under-risking of portfolios
		\item Return momentum and mean reversion
		\item Market volatility driven by psychological framing
	\end{itemize}
	
	\vspace{0.2cm}
	\tiny{\textcite{thaler1990mental}}
\end{frame}

\subsubsection{Independent Deviations}
\begin{frame}{Correlated Biases vs. Independent Deviations}
	
	\begin{block}{The EMH Assumption}
		Behavioral deviations are assumed to be \textbf{random and uncorrelated}, and thus cancel out in aggregate.
	\end{block}
	
	\vspace{0.3cm}
	\textbf{Psychological Evidence Suggests Otherwise:}
	\begin{itemize}
		\item \textbf{Representativeness Bias}:
		\begin{itemize}
			\item Investors draw broad conclusions from \textcolor{red}{limited or recent data}.
			\item Leads to overreaction -- e.g., price bubbles and excessive optimism.
		\end{itemize}
		
		\item \textbf{Conservatism Bias}:
		\begin{itemize}
			\item Investors are slow to update beliefs after receiving new information.
			\item Leads to underreaction -- e.g., price drifts after earnings surprises.
		\end{itemize}
	\end{itemize}
	\begin{alertblock}{Conclusion}
		Behavioral biases are \textbf{systematic} and often \textbf{correlated}, challenging the EMH view that errors cancel out.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Empirical Evidence: Crashes and Bubbles}
	
	\begin{block}{Challenge to the “Independent Errors” Assumption}
		Large-scale market movements not explained by fundamentals suggest correlated investor behavior.
	\end{block}
	
	\vspace{0.3cm}
	\begin{itemize}
		\item \textbf{Crash Example — Black Monday (1987)}:
		\begin{itemize}
			\item On October 19, 1987, the stock market dropped by 20–25\% in a single day.
			\item No significant news preceded the drop — inconsistent with EMH.
			\item Suggests panic selling and herding behavior.
		\end{itemize}
		
		\item \textbf{Bubble Example — Dot-Com Boom (late 1990s)}:
		\begin{itemize}
			\item Investors overreacted to limited information about tech firms.
			\item Price increases fed representativeness bias and trend-chasing.
			\item Prices collapsed when fundamentals failed to catch up.
		\end{itemize}
	\end{itemize}
	
	\begin{alertblock}{Conclusion}
		Crashes and bubbles are macro-level symptoms of correlated behavioral deviations — not independent noise.
	\end{alertblock}
	
\end{frame}

\subsubsection{Arbitrage}
\begin{frame}{Arbitrage and Its Limitations}
	
	\begin{block}{Theoretical Role of Arbitrage}
		In efficient markets, mispricings are quickly corrected by arbitrageurs who exploit price discrepancies.
	\end{block}
	
	\vspace{0.2cm}
	\textbf{Example: Shorting an Overpriced Stock}
	\begin{itemize}
		\item Identify firm ABC as overpriced based on fundamentals.
		\item Short the stock — profit when price reverts to fair value.
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Why Arbitrage May Fail in Reality}
	\begin{itemize}
		\item Mispricing may persist longer than arbitrageurs can remain solvent.
		\item Arbitrage is risky, capital-intensive, and often constrained (e.g., short-sale restrictions).
		\item Noise traders can push prices further from fundamentals in the short run.
	\end{itemize}
	
	\vspace{0.3cm}
	\tiny{\textcite{shleifer1997limits}, Keynes: "Markets can remain irrational longer than you can remain solvent."}
\end{frame}

\begin{frame}{Empirical Evidence: Limits to Arbitrage}
	
	\begin{itemize}
		\item \textbf{Quote}: “Markets can stay irrational longer than you can stay solvent.” — \textit{John Maynard Keynes}
		
		\item \textbf{Earnings Surprises}
		\begin{itemize}
			\item Prices adjust gradually — evidence of underreaction
			\item Violates EMH's assumption of rapid correction
		\end{itemize}
		
		\item \textbf{Small Cap Anomaly}
		\begin{itemize}
			\item Small-cap stocks tend to earn abnormally high returns
			\item Illiquidity and arbitrage frictions prevent quick correction
		\end{itemize}
		
		\item \textbf{Value vs. Growth}
		\begin{itemize}
			\item High B/P or E/P stocks outperform growth stocks
			\item Contradicts EMH; may persist due to style preference or constraints
		\end{itemize}
	\end{itemize}
	
	\vspace{0.2cm}
	\begin{alertblock}{Conclusion}
		When arbitrage is risky, costly, and delayed, pricing errors may persist — violating the strong-form EMH.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Challenges to the Efficient Market Hypothesis}
	
	\begin{block}{Three Foundational Assumptions — All Challenged}
		\begin{itemize}
			\item \textbf{Rationality}: Investors are not always rational (overconfidence, disposition effect, framing bias)
			\item \textbf{Independent Deviations}: Biases are correlated and persistent (herding, bubbles)
			\item \textbf{Unlimited Arbitrage}: Real-world frictions limit arbitrage (risk, cost, timing)
		\end{itemize}
	\end{block}
	
	\vspace{0.4cm}
	\begin{alertblock}{Implication}
		Market mispricing may persist — challenging the view that prices always and fully reflect all available information.
	\end{alertblock}
	
\end{frame}

\begin{frame}{Critiques of Behavioral Finance}

	\begin{itemize}
		\item \textbf{File Drawer Problem}:
		\begin{itemize}
			\item Studies that fail to reject EMH often remain unpublished.
			\item Journals may favor “novel” findings
			\item This publication bias may exaggerate the perceived failure of EMH.
		\end{itemize}
		
		\item \textbf{Failure to Account for Risk}:
		\begin{itemize}
			\item Some “anomalies” may reflect unmodeled risk factors.
			\item “Irrational” behavior may actually be compensation for hidden or behavioral risk.
		\end{itemize}
		
		\item \textbf{Weak Price Implications}:
		\begin{itemize}
			\item Behavioral theories often focus on belief biases...
			\item ...but do not always produce precise, testable predictions for asset prices.
		\end{itemize}
	\end{itemize}
	
	\begin{alertblock}{Bottom Line}
		Behavioral finance challenges the EMH — but may overstate its case due to selective evidence and interpretive ambiguity.
	\end{alertblock}
	
\end{frame}


\begin{frame}{Why EMH Still Matters}
	
	\begin{block}{The Role of EMH in Financial Thinking}
		\begin{itemize}
			\item \textbf{Benchmark}: Provides a null hypothesis for asset pricing, valuation, and regulation.
			\item \textbf{Discipline}: Forces clear specification of when and how markets deviate.
			\item \textbf{Empirical support}: Many anomalies fade after publication; markets are often efficient enough.
		\end{itemize}
	\end{block}
	
	\vspace{0.4cm}
	\begin{alertblock}{A Balanced View}
		While EMH may not fully explain all market outcomes, it remains a powerful baseline — especially when combined with insights from behavioral finance.
	\end{alertblock}
	
\end{frame}


\subsection{Implications for Corporate Finance}
\begin{frame}{Implications of EMH for Corporate Finance}
	
	\begin{block}{I. Managerial Constraints Under EMH}
		\begin{itemize}
			\item \textbf{Timing is futile (Strong-form EMH)}  
			\begin{itemize}
				\item Managers cannot exploit public or private information to time equity or debt issuance.
			\end{itemize}
			
			\item \textbf{Accounting changes don't matter (Semi-strong EMH)}  
			\begin{itemize}
				\item Cosmetic changes in accounting do not affect firm value.
				\item Only value-relevant disclosures move prices.
			\end{itemize}
		\end{itemize}
	\end{block}

	
	\begin{block}{II. Market Prices as Signals}
		\begin{itemize}
			\item \textbf{Prices are informative}  
			\begin{itemize}
				\item Managers can rely on market prices as a summary of dispersed investor expectations.
			\end{itemize}
			
			\item \textbf{Speculation is discouraged}  
			\begin{itemize}
				\item Currency and commodity speculation is unlikely to yield persistent excess returns.
			\end{itemize}
		\end{itemize}
	\end{block}
	
\end{frame}

\subsection*{Summary}
\begin{frame}{Summary: Efficient Markets and Behavioral Challenges}
	
	\begin{block}{1. Can Financing Decisions Create Value?}
		\begin{itemize}
			\item Yes — via market timing, tax arbitrage, or security innovation
			\item But success depends on market efficiency and persistence of mispricing
		\end{itemize}
	\end{block}
	
	\begin{block}{2. EMH — Theory and Empirical Support}
		\begin{itemize}
			\item Weak-form: Prices follow a random walk
			\item Semi-strong: Prices incorporate public news quickly (event studies)
			\item Strong-form: Rarely holds — insiders can earn abnormal returns
		\end{itemize}
	\end{block}
	

	\begin{block}{3. Behavioral Challenges to EMH}
		\begin{itemize}
			\item Psychological biases: Overconfidence, disposition, home bias
			\item Correlated misbeliefs: Crashes, bubbles, herding
			\item Limited arbitrage: Risk, constraints, slow correction
		\end{itemize}
	\end{block}
	
\end{frame}
